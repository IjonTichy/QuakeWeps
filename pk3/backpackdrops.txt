actor BackpackDrop: CustomInventory
{
    inventory.maxamount 0
    +AUTOACTIVATE

    +FLOATBOB
    Scale 0.9

    States
    {
      Spawn:
        QPAK A -1 bright
        stop
    }
}

actor ZombiemanBackpackDrop: BackpackDrop
{
    Inventory.PickupMessage "You got 2 shells, 5 nails"

    States
    {
      Pickup:
        TNT1 A 0 A_GiveInventory("QuakeShell", 2)
        TNT1 A 0 A_GiveInventory("QuakeNails", 5)
        stop
    }
}
