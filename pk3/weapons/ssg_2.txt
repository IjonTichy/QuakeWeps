actor "Quake 2 SSG": Weapon
{
    Weapon.Kickback 100
    Weapon.SelectionOrder 5
    Weapon.AmmoUse 3
    Weapon.AmmoGive 9
    Weapon.AmmoType "QuakeShell"

    Weapon.SlotNumber 3

    +NOAUTOAIM
    +FLOATBOB

    Inventory.Icon "WSSGA0"
    Inventory.PickupMessage "You got the second double-barreled shotgun"
    Obituary "%o eats two heavy loads of %k's buckshot"

    States
    {
      Deselect:
        TNT1 AAAAAAAAAAAAAAAAAAA 0 A_Lower
        SHT2 A 1 A_Lower
        Loop

      Select:
        TNT1 AAAAAAAAAAAAAAAAAAA 0 A_Raise
        SHT2 A 1 A_Raise
        Loop

      Ready:
        SHT2 A 1 A_WeaponReady
        Loop

      Fire:
        SHTG A 0 A_FireBullets(14, 14, 30, 2, "BulletPuff")
        SHT2 A 0 A_PlaySound ("weapons/sshotf", CHAN_WEAPON)
        SHT2 A 0 A_GunFlash
        SHT2 A 2 bright offset(0, 40)
        SHT2 A 2 bright offset(0, 42)
        SHT2 A 2 bright offset(0, 43)
        SHT2 A 2 bright offset(0, 42)
        SHT2 A 2 offset(0, 40)
        SHT2 A 2 offset(0, 37)
        SHT2 A 2 offset(0, 33)
        SHT2 A 2 offset(0, 32)
        PKS2 B 2 A_PlaySoundEx("weapons/sshoto", "Weapon")
        PKS2 C 2 offset(0, 33)
        PKS2 D 2 offset(0, 35)
        PKS2 E 2 offset(0, 38)
        PKS2 E 2 offset(0, 37)
        PKS2 E 2 offset(0, 35)
        PKS2 D 2 offset(0, 35) A_PlaySoundEx("weapons/sshotc", "Weapon")
        PKS2 C 2 offset(0, 34)
        PKS2 B 2 offset(0, 33)
        PKS2 A 2 offset(0, 32)
        PKS2 A 2 offset(0, 33)
        PKS2 A 2 offset(0, 32)
        SHT2 A 0 A_ReFire
        Goto Ready

      Flash:
        SHT2 I 4 Bright A_Light1
        SHT2 J 3 Bright A_Light2
        Goto LightDone
      Spawn:
        WSSG A -1 bright
        Stop
    }
}
